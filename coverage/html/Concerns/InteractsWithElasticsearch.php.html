597    /**
598     * Wait for Elasticsearch to be healthy and ready.
599     *
600     * Polls the Elasticsearch container health check until it reports
601     * healthy status. Uses docker inspect to check container health.
602     *
603     * Polling strategy:
604     * - Maximum attempts: 60 (configurable)
605     * - Delay between attempts: 2 seconds
606     * - Total maximum wait: 120 seconds
607     *
608     * Health check:
609     * - Uses Elasticsearch _cluster/health API
610     * - Checks for status: green or yellow
611     * - Container must be running and healthy
612     *
613     * Why this is needed:
614     * - Elasticsearch takes time to start (30-60 seconds)
615     * - Application will fail if Elasticsearch not ready
616     * - Health check ensures cluster is operational
617     *
618     * @param  string $appPath     Absolute path to application directory
619     * @param  string $serviceName Name of service in docker-compose.yml
620     * @param  int    $maxAttempts Maximum number of polling attempts
621     * @return bool   True if Elasticsearch is healthy, false if timeout
622     */
623    protected function waitForElasticsearchHealth(string $appPath, string $serviceName, int $maxAttempts = 60): bool
624    {
625        $attempts = 0;
626
627        while ($attempts < $maxAttempts) {
628            // Check container health status using docker inspect
629            $output = $this->process()->run(
630                ['docker', 'compose', 'ps', '--format', 'json', $serviceName],
631                $appPath
632            );
633
634            if ($output !== null) {
635                $output = trim($output);
636                if ($output !== '') {
637                    // Parse JSON output to check health status
638                    $serviceInfo = json_decode($output, true);
639                    if (is_array($serviceInfo) && isset($serviceInfo['Health']) && $serviceInfo['Health'] === 'healthy') {
640                        return true;
641                    }
642
643                    // Fallback: Try to curl the health endpoint
644                    if ($this->process()->succeeds(['curl', '-f', 'http://localhost:9200/_cluster/health'])) {
645                        return true;
646                    }
647                }
648            }
649
650            // Wait 2 seconds before next attempt
651            sleep(2);
652            $attempts++;
653        }
654
655        return false;
656    }
